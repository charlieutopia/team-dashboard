+import { Router } from 'express';
+import { db } from './db';
+
+export interface Order {
+  id: string;
+  customer_id: string;
+  status: 'draft' | 'confirmed' | 'fulfilled' | 'cancelled';
+  line_items: LineItem[];
+  created_at: Date;
+  notes?: string;
+}
+
+export interface LineItem {
+  product_id: string;
+  quantity: number;
+  unit_price: number;
+  product_name: string;
+  sku: string;
+}
+
+const router = Router();
+
+router.post('/api/orders', async (req, res) => {
+  const { customer_id, cart_id, notes } = req.body;
+
+  const cart = await db.getCart(cart_id);
+  if (!cart || cart.items.length === 0) {
+    return res.status(400).json({ error: 'Invalid cart' });
+  }
+
+  const order: Order = {
+    id: `ord_${Date.now()}`,
+    customer_id,
+    status: 'draft',
+    line_items: cart.items.map(item => ({
+      product_id: item.product_id,
+      quantity: item.quantity,
+      unit_price: item.price,
+      product_name: item.name,
+      sku: item.sku,
+    })),
+    created_at: new Date(),
+    notes,
+  };
+
+  await db.createOrder(order);
+  res.status(201).json(order);
+});
+
+export default router;

+export async function convertCartToOrder(cartId: string): Promise<Order> {
+  const cart = await db.getCart(cartId);
+
+  if (!cart) {
+    throw new Error('Cart not found');
+  }
+
+  // Reserve inventory for each line item
+  for (const item of cart.items) {
+    const available = await db.checkInventory(item.product_id);
+    if (available < item.quantity) {
+      throw new Error(`Insufficient stock for product ${item.product_id}`);
+    }
+  }
+
+  // Create order from cart
+  const order = await db.createOrderFromCart(cart);
+  await db.clearCart(cartId);
+
+  return order;
+}
