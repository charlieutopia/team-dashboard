+export async function convertCartToOrder(
+  cart_id: string,
+  idempotency_key: string,
+): Promise<Order> {
+  // Check idempotency: if order with this key exists, return it
+  const existing = await db.orders.findOne({ idempotency_key });
+  if (existing) return existing;
+
+  // Validate cart state
+  const cart = await db.carts.findById(cart_id);
+  if (!cart || cart.is_converted) {
+    throw new Error("Cart already converted or not found");
+  }
+
+  // Reserve inventory
+  for (const item of cart.items) {
+    const stock = await db.inventory.reserve(item.product_id, item.quantity);
+    if (!stock) throw new Error(`Insufficient stock for ${item.product_id}`);
+  }
+
+  // Create order with line items
+  const order = await db.orders.insert({
+    customer_id: cart.customer_id,
+    cart_id,
+    status: "draft",
+    idempotency_key,
+    line_items: cart.items,
+    created_at: new Date().toISOString(),
+  });
+
+  // Clear the cart
+  await db.carts.update(cart_id, { is_converted: true, cleared_at: new Date().toISOString() });
+  return order;
 }
